To Run
# 1. Copy and fill secrets
cp .env.example .env

# 2. Boot node (identity + Docker stack)
python boot.py

# 3. Run tests
pytest tests/ -v

# 4. Start registry
cd registry && uvicorn registry.main:app --port 8000

# 5. Deploy contract (after setting WALLET_PRIVATE_KEY in .env)
python contracts/deploy.py

# 6. Start blockchain event listener
python -m registry.event_listener
